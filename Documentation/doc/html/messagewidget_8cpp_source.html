174 void MessageWidget::paintEvent( QPaintEvent* event )
175 {
176  Q_UNUSED( event )
177 
178  QPainter painter{ this };
179 
180  constexpr float radius = 4 * 0.6;
181  const QRect innerRect = rect().marginsRemoved( QMargins() + borderSize / 2 );
182  const QColor color = palette().color( QPalette::Window );
183  constexpr float alpha = 0.2;
184  const QColor parentWindowColor =
185  ( parentWidget() ? parentWidget()->palette() : qApp->palette() )
186  .color( QPalette::Window );
187  const int newRed = ( color.red() * alpha )
188  + ( parentWindowColor.red() * ( 1 - alpha ) );
189  const int newGreen = ( color.green() * alpha )
190  + ( parentWindowColor.green() * ( 1 - alpha ) );
191  const int newBlue = ( color.blue() * alpha )
192  + ( parentWindowColor.blue() * ( 1 - alpha ) );
193  painter.setPen( QPen( color, borderSize ) );
194  painter.setBrush( QColor( newRed, newGreen, newBlue ) );
195  painter.setRenderHint( QPainter::Antialiasing );
196  painter.drawRoundedRect( innerRect, radius, radius );
197 }
